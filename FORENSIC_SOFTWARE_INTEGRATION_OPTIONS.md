from fastapi import UploadFile
from pathlib import Path
import subprocess

MOBILEDIT_PATH = "/path/to/mobiledit/bin"

async def request_forensic_analysis(
    client_number: str,
    case_id: str,
    consent_document: str
):
    """
    Create forensic analysis request
    """
    analysis = {
        "analysis_id": generate_analysis_id(),
        "client_number": client_number,
        "case_id": case_id,
        "consent_document": consent_document,
        "status": "pending",
        "created_at": datetime.utcnow(),
        "estimated_completion": datetime.utcnow() + timedelta(days=3)
    }
    
    await db.forensic_analyses.insert_one(analysis)
    
    # Send notification to forensic analyst
    await send_analyst_notification(analysis)
    
    return analysis

async def upload_forensic_report(
    analysis_id: str,
    report_file: UploadFile
):
    """
    Upload completed MOBILedit report
    """
    # Save report
    report_path = FORENSIC_REPORTS_DIR / f"{analysis_id}.pdf"
    
    with open(report_path, "wb") as f:
        content = await report_file.read()
        f.write(content)
    
    # Update database
    await db.forensic_analyses.update_one(
        {"analysis_id": analysis_id},
        {"$set": {
            "status": "completed",
            "report_path": str(report_path),
            "completed_at": datetime.utcnow()
        }}
    )
    
    # Notify client & lawyer
    await send_completion_notification(analysis_id)
    
    return {"success": True, "report_path": str(report_path)}
